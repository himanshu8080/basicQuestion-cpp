...................factorial using recursion ......................

#include <iostream>
using namespace std;

int factorial(int n){
    
    if(n==0)
    return 1;
    
    int smallerproblem=factorial(n-1);
    int biggerproblem=n*smallerproblem;
    
    return biggerproblem;
    

}


int main() {
    int n ;
    cin>>n;
    
   int ans=factorial(n);
   cout <<ans<<endl;
    return 0;
}

#include <iostream>
using namespace std;

int power(int n){
    
    if(n==0)
    return 1;
    
    return 2*power(n-1);

}


int main() {
    int n ;
    cin>>n;
    
   int ans=power(n);
   cout <<ans<<endl;
    return 0;
}
